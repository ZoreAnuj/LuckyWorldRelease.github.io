Start Using Lucky Robots

Get immediate access to the platform. Install Lucky Robots in minutes, and start building, training, and experimenting with robotic AI models today.

# Install LuckyWorld
pip install luckyworld

# Import and initialize your robot
from luckyworld import Robot, Environment

# Create environment and robot
env = Environment("home_scene")
robot = Robot("SO100", environment=env)

# Use natural language commands
robot.execute_command("locate my keyboard")
robot.execute_command("navigate to the kitchen")

# Record training episodes
episodes = robot.record_episodes(num_episodes=50)

# Train your policy
policy = robot.train_policy(episodes)

# Run inference with trained model
robot.execute_policy(policy, task="find and bring me my coffee mug")
